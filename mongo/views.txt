// 1. Se debe realizar una vista que devuelva las facturas ordenadas por fecha
db.createView(
    "facturas_por_fecha",
    "E01_FACTURA",
    [
        {
            $sort: {fecha: 1}
        },
        {
            $project: {
                detalles_factura: 0
            }
        }
    ]
);


// 2. Se necesita una vista que devuelva todos los productos que a√∫n no han sido facturados.
db.createView(
    "productos_no_facturados",
    "E01_PRODUCTO",
    [
       {
            $lookup: {
                from: "E01_FACTURA",
                localField: "_id",
                foreignField: "detalles_factura.codigo_producto",
                as: "facturas"
            }
       },
       {
            $match: {
                facturas: { $eq: [] }
            }
       },
       {
            $project: {
                facturas: 0
            }
       }
    ]
);
